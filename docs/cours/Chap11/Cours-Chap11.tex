\subsection{Le cercle trigonométrique}

\begin{itemize}
\item[$\bullet$] Le cercle trigonométrique est le cercle de rayon 1, orienté dans le sens inverse des aiguilles d'une montre
\item[$\bullet$]La circonférence du cercle trigonométrique est de $2\pi$
\item[$\bullet$]En enroulant la droite des réels sur un cercle trigonométrique, on fait correspondre tout nombre réel à un point du cercle, nommé point-image.
\item[$\bullet$] En enroulant la droite des réels sur le cercle trigonométrique, tout point du cercle est le point-image d'une infinité de réel.

\bigskip

\textbf{Exemple : } :\href{https://www.geogebra.org/m/Ry4j9ZT2}{animation} 

\bigskip
\item[$\bullet$]Soit M un point du cercle trigonométrique, repéré par le nombre $x$. $x$ est la mesure en radian de l'angle orienté $(\vec{i},\overrightarrow{OM})$.\\
$x,x+2\pi,x+4\pi,...$ et $x-2\pi,x-4\pi,x-6\pi$ repèrent le même point M sur le cercle.\\
 $(\vec{i},\overrightarrow{OM})=x[2\pi]$("modulo $2\pi$")
 
\item[$\bullet$]La mesure principale d'un angle orienté est son unique mesure appartenant à l'intervalle $]-\pi,\pi]$

\end{itemize}
\bigskip

\cadreR{
\textbf{Méthode :}
Soit $\dfrac{a\pi}{b}$ où $a$ et $b$ sont deux entiers avec $b\neq 0$ dont on veut trouver la mesure principale.\\

Pour cela, on cherche un multiple de $2b$ le plus proche de $a$ en excès ou en défaut, et on décompose $\dfrac{a\pi}{b}$.

}

\ex 


$\dfrac{50}{3}$ \\
$16\times 3 =48 $ est le plus proche multiple de $2b$ proche de $a$. En effet $18 \times 3=54$ est moins pertinent.
\\
$\dfrac{50}{3}=\dfrac{48\pi}{3}+\dfrac{2\pi}{3}=\dfrac{2\pi}{3}+8\times 2\pi=\dfrac{2\pi}{3}[2\pi]$. \\
$\dfrac{2\pi}{3}$ est la mesure principale.

\bigskip

\cadreR{
\begin{defi}
On définit le sinus et le cosinus d'un nombre réel à partir du cercle trigonométrique.\\
Soit $x$ un réel et son point-image M sur le cercle trigonométrique :
\begin{itemize}
\item[$\bullet$]$\cos(x)$ est l'abscisse de M.
\item[$\bullet$]$\sin(x)$ est l'ordonnée de M.
\item[$\bullet$] Pour tout $x\in \R; -1\leq \cos(x)\leq 1$
\item[$\bullet$] Pour tout $x\in \R; -1\leq \sin(x)\leq 1$
\item[$\bullet$] Pour tout $x\in \R; \cos^2(x)+\sin^2(x) =1$
\item[$\bullet$] Pour tout $x\in \R; \cos(x+2\pi)=\cos(x)$ et $\sin(x+2\pi)=\sin(x)$ 
\item[$\bullet$] Pour tout $x\in \R; \cos(-x)=\cos(x)$ et $\sin(-x)=-\sin(x)$ 
\item[$\bullet$] Pour tout $x\in \R; \cos(x+\pi)=-\cos(x)$ et $\sin(x+\pi)=-\sin(x)$ 
\item[$\bullet$] Pour tout $x\in \R; \cos(\pi-x)=-\cos(x)$ et $\sin(\pi-x)=\sin(x)$ 

\end{itemize}
\end{defi}

}

\subsection{Formules d'addition et de duplication}



\cadreR{
\begin{pte}

On se place dans un repère orthonormé du plan. Soient $\vec{u}=\begin{pmatrix}
x\\
y
\end{pmatrix}$ et $\vec{v}=\begin{pmatrix}
x'\\
y'
\end{pmatrix}$.
Le produit scalaire de deux vecteurs peut s'obtenir de deux façons suivantes :

$$\vec{u}.\vec{v}=xx'+yy'=\lVert\vec{u}\rVert \times \lVert\vec{v}\rVert \times \cos(\vec{u},\vec{v})$$ 
\end{pte}
}


\bigskip

\cadreR{
\begin{pte}
Soient $a$ et $b$ deux nombres réels.
\textbf{Formule d'addition
}
\begin{itemize}
\item[$\bullet$]$\cos(a+b)=\cos(a)\cos(b)-\sin(a)\sin(b)$
\item[$\bullet$]$\cos(a-b)=\cos(a)\cos(b)+\sin(a)\sin(b)$
\item[$\bullet$]$\sin(a+b)=\sin(a)\cos(b)+\cos(a)\sin(b)$
\item[$\bullet$]$\sin(a-b)=\sin(a)\cos(b)-\cos(a)\sin(b)$
\end{itemize}


Conséquence : \\

Comme pour tout $x\in \R; \cos^2(x)+\sin^2(x) =1$, on obtient $\cos^2(x)=1-\sin^2(x)$ ou $\sin^2(x)=1-\cos^2(x)$
\\
On en déduit \textbf{les formules de duplication} :\\
$\cos(2a)=\cos^2(a)-\sin^2(a)=\cos^2(a)-(1-\cos^2(a))=2\cos^2(a)-1$
\\
ou bien 
\\
 $\cos(2a)=\cos^2(a)-\sin^2(a)=1-\sin^2(a)-\sin^2(a)=1-2\sin^2(a)$

\end{pte}

}
\bigskip

\textbf{Explications :}
\begin{center}

\includegraphics[scale=0.8]{formules-addition.eps}

\end{center}

$\overrightarrow{OA}=\begin{pmatrix}
\cos(a)\\
\sin(a)
\end{pmatrix}$ et $\overrightarrow{OB}=\begin{pmatrix}
\cos(b)\\
\sin(b)
\end{pmatrix}$

Donc le produit scalaire $\overrightarrow{OB}.\overrightarrow{OA}=\cos(a)\cos(b)+\sin(a)\sin(b)$
\\

et d'autre part \\

$\overrightarrow{OB}.\overrightarrow{OA}=\lVert\overrightarrow{OA}\rVert \times \lVert\overrightarrow{OB} \rVert \times \cos(\overrightarrow{OB}, \overrightarrow{OA})=1\times 1\times \cos(a-b)$\\
Ainsi, on voit que $\cos(a-b)=\cos(a)\cos(b)+\sin(a)\sin(b)$

\bigskip

\ex 

\begin{enumerate}

\item Démontrer que $\dfrac{5\pi}{12}=\dfrac{\pi}{4}+\dfrac{\pi}{6}$
\item En déduire $\cos(\dfrac{5\pi}{12})$ et $\sin(\dfrac{5\pi}{12})$.


\end{enumerate}

\begin{enumerate}
\item $\dfrac{\pi}{4}+\dfrac{\pi}{6}=\dfrac{3\pi}{12}+\dfrac{2\pi}{12}=\dfrac{5\pi}{12}$
\item d'après les formules de l'addition :
\\
$\cos(\dfrac{5\pi}{12})=\cos(\dfrac{\pi}{4}+\dfrac{\pi}{6})=\cos(\dfrac{\pi}{4})\cos(\dfrac{\pi}{6})-\sin(\dfrac{\pi}{4})\sin(\dfrac{\pi}{6})=\dfrac{\sqrt{2}}{2}\times \dfrac{\sqrt{3}}{2}-\dfrac{\sqrt{2}}{2}\times \dfrac{1}{2}=\\\dfrac{\sqrt{6}}{4}-\dfrac{\sqrt{2}}{4}=\dfrac{\sqrt{6}-\sqrt{2}}{4}$\\

$\sin(\dfrac{5\pi}{12})=\sin(\dfrac{\pi}{4}+\dfrac{\pi}{6})=\sin(\dfrac{\pi}{4})\cos(\dfrac{\pi}{6})+\cos(\dfrac{\pi}{4})\sin(\dfrac{\pi}{6})=\dfrac{\sqrt{2}}{2}\times\dfrac{\sqrt{3}}{2}+\dfrac{\sqrt{2}}{2}\times \dfrac{1}{2}=\dfrac{\sqrt{6}+\sqrt{2}}{4}$


\end{enumerate}

\subsection{Equations- Inéquations trigonométriques}
\cadreR{

\begin{pte}
Soient $a$ et $x$ deux nombres réels. 
\begin{itemize}
\item[$\bullet$]$\cos(x)=\cos(a)\Leftrightarrow x=a+2k\pi$ ou $x=-a+2k\pi; k\in \Z$
\item[$\bullet$]$\sin(x)=\sin(a)\Leftrightarrow x=a+2k\pi$ ou $x=\pi-a+2k\pi; k\in \Z$
\end{itemize}

\end{pte}
}

\item $3-2\sqrt{3}\cos(x)<0$
\item $2\cos(x)+\sqrt{3}<0$
\item $1-2\sin(x)>0$
\end{enumerate}

\subsection{Restriction du domaine d'étude d'une fonction trigonométrique}

\cadreR{
\begin{defi}
\begin{enumerate}
\item Une fonction $f$ est paire si pour tout $x\in \R, f(-x)=f(x)$
\item Une fonction $f$ est impaire si pour tout $x\in \R, f(-x)=-f(x)$
\item Une fonction $f$ est périodique de période $T$ si pour tout $x\in \R, f(x+T)=f(x)$
\end{enumerate}

\end{defi}
}

\bigskip

\cadreR{
\begin{pte}

\begin{enumerate}

\item Si une fonction $f$ est paire alors $\mathcal{C}_f$ présente une symétrie selon l'axe des abscisses.
\item Si une fonction $f$ est impaire alors $\mathcal{C}_f$ présente une symétrie centrale de centre O.
\item Si $f$ est $T-$périodique alors $\mathcal{C}_f$ est invariante par la translation de vecteur $T\vec{i}$
\end{enumerate}
\end{pte}
}

\bigskip
\begin{rem}
\textbf{Conséquences :}
\begin{enumerate}
\item Si la fonction $f$ est paire ou impaire, on peut l'étudier sur $\R^+$
\item Si $f$ est une fonction $T-$périodique, on peut l'étudier sur tout intervalle d'amplitude $T$.
\item Si $f$ est une fonction paire ou impaire de période $T$, on peut l'étudier sur l'intervalle $[0,\dfrac{T}{2}$
\end{enumerate}
\end{rem}

\subsection{Fonctions cosinus et sinus}

\cadreR{ 
\begin{pte}
\begin{enumerate}
\item $\cos(-x)=\cos(x)$ donc la fonction $\cos$ est paire sur $\R$.
\item $\sin(-x)=-\sin(x)$ donc la fonction $\sin$ est impaire sur $\R$.
\item $\cos(x+2\pi)=\cos(x)$donc la fonction $\cos$ est $2\pi$-périodique.
\item $\sin(x+2\pi)=\sin(x)$donc la fonction $\sin$ est $2\pi$-périodique.

\end{enumerate}

\end{pte}

}

\subsubsection{Dérivées}

\cadreR{
\begin{pte}
\begin{enumerate}
\item La fonction sinus est dérivable sur $\R$ et pour tout réel $x, sin'(x)=cos(x)$.
\item La fonction cosinus est dérivable sur $\R$ et pour tout réel $x, cos'(x)=-sin(x)$
\end{enumerate}

\end{pte}
}

\bigskip
\begin{proof}



\textbf{pré-requis :} \\
$\displaystyle\lim_{x\to 0}\dfrac{\sin(x)}{x}=1$ et $\displaystyle\lim_{x\to 0}\dfrac{\cos(x)-1}{x}=0$

\bigskip
$\dfrac{\sin(a+h)-\sin(a)}{h}=\dfrac{\sin(a)\cos(h)+\cos(a)\sin(h)-\sin(a)}{h}=\dfrac{\sin(a)(\cos(h)-1)}{h}+\dfrac{\cos(a)\sin(h)}{h}$\\

$\dfrac{\sin(a+h)-\sin(a)}{h}=\sin(a)\dfrac{\cos(h)-1}{h}+\cos(a)\dfrac{\sin(h)}{h}$ \\

D'après les pré-requis, on a $\displaystyle\lim_{h\to 0}\dfrac{\sin(h)}{h}=1$ et $\displaystyle\lim_{h\to 0}\dfrac{\cos(h)-1}{h}=0$. \\
Donc $\displaystyle\lim_{h\to 0}\dfrac{\sin(a+h)-\sin(a)}{h}=\sin(a)\times 0+\cos(a)\times 1=\cos(a)$.
Donc la fonction sinus est dérivable et $\sin'(x)=\cos(x)$.

\bigskip

Pour tout réel, $\cos(x)=\sin(\frac{\pi}{2}-x)$ donc par composition, $\cos'(x)=-1\times \sin'(\frac{\pi}{2}-x)=-\cos(\frac{\pi}{2}-x)=-\sin(x)$.
\end{proof}
